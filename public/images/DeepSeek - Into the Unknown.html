      price: car.price || "",
      ratings: car.ratings || "",
    });

    // Set the search queries for company and color
    if (car.company) {
      const selectedCompany = companies.find((c) => c.id === car.company);
      if (selectedCompany) {
        setSearchCompanyQuery(selectedCompany.title);
      }
    }

    if (car.color) {
      setSearchColorQuery(car.color);
    }
  }
}, [car, companies]); // Add `companies` to the dependency array

Explanation

  1. Populate searchCompanyQuery:

    • When the car prop is available, find the corresponding company from the companies list using the company ID stored in car.

    • If the company is found, set the searchCompanyQuery state to the company's title.

  2. Populate searchColorQuery:

    • If the car has a color value, set the searchColorQuery state to that color.

  3. Dependency Array:

    • Add companies to the dependency array of the useEffect to ensure the company name is correctly populated when the companies list is fetched.
